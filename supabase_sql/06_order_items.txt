-- Create the order_items table
CREATE TABLE public.order_items (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id uuid NOT NULL REFERENCES public.orders(id) ON DELETE CASCADE,
  product_id_ref uuid REFERENCES public.products(id) ON DELETE SET NULL, -- Reference to the original product
  product_name text NOT NULL, -- Snapshot of product name at time of order
  quantity integer NOT NULL,
  price_at_purchase numeric(10, 2) NOT NULL, -- Price per unit at time of purchase
  selected_size text NOT NULL,
  created_at timestamptz DEFAULT now() NOT NULL
  -- No updated_at needed usually for order items as they are typically immutable post-creation
);

-- Enable Row Level Security
ALTER TABLE public.order_items ENABLE ROW LEVEL SECURITY;

-- Add Indexes
CREATE INDEX IF NOT EXISTS idx_order_items_order_id ON public.order_items (order_id);
CREATE INDEX IF NOT EXISTS idx_order_items_product_id_ref ON public.order_items (product_id_ref);
